[Desktop Entry]
Type=Application
Name=Limit CPU Usage
Comment=Limit CPU Usage if the package temperature exceeds 60 degrees Celcius.
Icon=thermal-monitor
Exec=sudo sh '/home/YOURUSERNAME/.local/share/applications/temp_throttle.sh' 60
Terminal=true
Categories=;
