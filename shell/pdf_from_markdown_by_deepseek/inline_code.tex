\usepackage{xcolor}
\usepackage{framed}
\usepackage{relsize}

% Define colors for inline code
\definecolor{codebg}{RGB}{240,240,240}
\definecolor{codeborder}{RGB}{200,200,200}

% Better inline code styling without recursion
\let\originaltexttt\texttt
\renewcommand{\texttt}[1]{%
    \colorbox{codebg}{%
        \originaltexttt{\textcolor{black}{#1}}%
    }%
}
